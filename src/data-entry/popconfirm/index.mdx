import { Playground, Props } from 'docz'

import Popconfirm from './popConfirm'

import {useState} from 'react'

# Popconfirm

The Popconfirm component is a tool to open a confirmation popover on any button or anchor that you would like.

## When to use 

It's most used for asking for user confirmation and it's more lightweight than the usual confirmation modal.

## Basic example

<Playground>
    {() => { 
        const [visible, setVisible] = useState(false);

        const showPopconfirm = () => {
            setVisible(!visible);
        };

        const confirm = (e) => {
            console.log(e, "success");
        };

        const cancel = (e) => {
            console.log(e, "cancel");
        };

        const position = {
            top:"20px",
            left:"50px"
        };
        const title = <p>Whatever you want to display</p>;
        const okText = "OK";
        const cancelText = "Cancel";
        const actionHandler = <a style={{cursor:'pointer'}} onClick={showPopconfirm}>Delete</a>
        

        return(
        <Popconfirm 
            title={title} 
            okText={okText} 
            cancelText={cancelText}
            containerWidth="400px"
            position={position}
            confirm={confirm}
            cancel={cancel}
            visible={visible}
            actionHandler={actionHandler}
        >
        </Popconfirm>
        )
    }}
</Playground>
